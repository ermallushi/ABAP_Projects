FORM add_it01.

  SELECT p~keyz1 p~posza p~selw1 p~selw2 p~selw3 p~betrz
         p~fikey p~bvrko p~gsber p~blart p~waers p~budat
         p~bldat p~opbel p~txtvw p~infof p~ruear p~txtrz p~klaeb
         p~aetim k~ernam k~aedat k~stazs k~chdsk
         j~offic
*    p~aetim
    INTO CORRESPONDING FIELDS OF TABLE it01
    FROM dfkkzp AS p
    JOIN dfkkzk  AS k ON p~keyz1 EQ k~keyz1
    LEFT JOIN dfkkcjt AS j ON p~keyz1 EQ j~prkey
    WHERE p~bldat IN s_bldat
    AND   p~budat IN s_budat
    AND   p~ruear IN s_ruear
    AND   p~txtrz IN s_txtrz
    AND   p~fikey IN s_fikey
    AND   p~opbel IN s_opbel

    AND   k~chdsk IN s_chdsk
    AND   k~keyz1 IN s_keyz1
    AND   k~keyz2 IN s_keyz2
    AND   k~infof IN s_infof
    AND   k~blart IN s_blart
    AND   k~ernam IN s_ernam
    AND   k~bvrko IN s_bvrko.

  LOOP AT it01.

    IF it01-offic NOT IN s_offic.
      DELETE it01.
      CONTINUE.
    ENDIF.

    READ TABLE gt_stext REFERENCE INTO gr_stext
      WITH KEY domvalue_l = it01-stazs BINARY SEARCH.

    IF sy-subrc EQ 0.
      it01-ddtext = gr_stext->ddtext.
    ENDIF.

    IF it01-klaeb IS NOT INITIAL.

      SELECT SINGLE gpart vkont
             FROM dfkkop
             INTO (it01-gpart, it01-vkont )
            WHERE opbel EQ it01-klaeb.
    ELSE.
      IF it01-selw1 IS NOT INITIAL.

        winv = it01-selw1.

        SELECT SINGLE gpart vkont "zphone_number
               INTO CORRESPONDING FIELDS OF it01
               FROM dfkkop
               WHERE xblnr EQ winv. "AND

        IF sy-subrc NE 0.
          it01-vkont = it01-selw2.
          it01-gpart = it01-selw3.
        ENDIF.
      ELSE.

        it01-vkont = it01-selw2.
        it01-gpart = it01-selw3.

        SELECT SINGLE xblnr gpart vkont "zphone_number
               FROM dfkkop
               INTO (it01-selw1, it01-gpart, it01-vkont )
              WHERE augbl EQ it01-opbel.

        SHIFT it01-selw1 LEFT DELETING LEADING '0'.

      ENDIF.
    ENDIF.

    SELECT SINGLE name_org1 name_first name_last source
           INTO CORRESPONDING FIELDS OF it01
           FROM but000
           WHERE partner EQ it01-gpart.

    REFRESH: itphone, lt_tdata.

    ls_tdata-vkont = it01-vkont.

    CALL FUNCTION 'IST_TDATA_SEARCH'
      EXPORTING
        is_ist_tdata   = ls_tdata
        iv_groupsearch = ''
      TABLES
        et_ist_tdata   = lt_tdata.

    LOOP AT lt_tdata.
      MOVE lt_tdata-telnum TO itphone-telnum.
      APPEND itphone.
    ENDLOOP.

    CLEAR wpn.
    SORT itphone.
    DELETE ADJACENT DUPLICATES FROM itphone.
    LOOP AT itphone.
      CONCATENATE wpn itphone-telnum
             INTO wpn SEPARATED BY ', '.
    ENDLOOP.
    SHIFT wpn BY 2 PLACES LEFT.
    it01-telnum = wpn.

    MODIFY it01.
  ENDLOOP.
ENDFORM. "add_it01

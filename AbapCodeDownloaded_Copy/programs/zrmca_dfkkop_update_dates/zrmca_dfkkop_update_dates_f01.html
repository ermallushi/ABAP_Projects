FORM fetch_data .
  DATA: lv_date_from TYPE datum,
        lv_date_to   TYPE datum.

* Fetch the contract account document items from dfkkop...
  SELECT opbel opupw opupk opupz budat abrzu abrzo blart
    FROM dfkkop
    INTO TABLE gt_dfkkop
    WHERE augst EQ space
    AND   opbel IN s_opbel
    AND   budat IN s_budat
    AND   blart IN s_blart.

  LOOP AT gt_dfkkop REFERENCE INTO gr_dfkkop.

* Set the first and last day of the month based on document posting date...

* Set first day of the posting month to lv_date_from...
    CONCATENATE gr_dfkkop->budat(6) gc_first_day INTO lv_date_from.

* Set last day of the posting month to lv_date_to...
    CALL FUNCTION 'RP_LAST_DAY_OF_MONTHS'
      EXPORTING
        day_in            = lv_date_from
      IMPORTING
        last_day_of_month = lv_date_to.

* Check whether first day of the posting month is equal to lower limit
* of settlement period (dfkkop-abrzu) and last day of the posting month
* is equal to upper limit of the billing period. If not, append the record
* to the ALV internal table...
    IF gr_dfkkop->abrzu NE lv_date_from OR
       gr_dfkkop->abrzo NE lv_date_to.

      ADD 1 TO gv_numb_of_rec.

      APPEND INITIAL LINE TO gt_items REFERENCE INTO gr_items.

      MOVE-CORRESPONDING gr_dfkkop->* TO gr_items->*.

      gr_items->abrzu_n = lv_date_from.
      gr_items->abrzo_n = lv_date_to.

      gs_dfkkko-opbel = gr_dfkkop->opbel.
      COLLECT gs_dfkkko INTO gt_dfkkko.
    ENDIF.
  ENDLOOP.

  SORT gt_items BY opbel opupw opupk opupz.
ENDFORM. " FETCH_DATA

FORM update_data.

* Update the documents...
  LOOP AT gt_dfkkko REFERENCE INTO gr_dfkkko.

* Lock the documents...
    CALL FUNCTION 'FKK_ENQ_OPBEL_FOR_CHANGES'
      EXPORTING
        i_opbel           = gr_dfkkko->opbel
      EXCEPTIONS
        foreign_lock      = 1
        system_failure    = 2
        foreign_lock_mass = 3
        OTHERS            = 4.

    CASE sy-subrc.
      WHEN '0'. " OK...
      WHEN '1' OR '3'. " Locked...
        PERFORM set_message USING 'L' gr_dfkkko->opbel.
        CONTINUE.
      WHEN OTHERS. " System failure...
        PERFORM set_message USING 'S' gr_dfkkko->opbel.
        CONTINUE.
    ENDCASE.

    PERFORM clear_parameters.

* Fetch the header data in open item accounting document
    CALL FUNCTION 'FKK_DOC_HEADER_SELECT_BY_OPBEL'
      EXPORTING
        i_opbel = gr_dfkkko->opbel
      IMPORTING
        e_fkkko = gs_fkkko
        e_count = gv_count.

    IF gv_count EQ 0.
      PERFORM set_message USING 'H' gr_dfkkko->opbel.
      CONTINUE.
    ENDIF.

    gv_count = 0.

* Fetch the G/L account items in open item account document
    CALL FUNCTION 'FKK_GL_LINE_ITEMS_SEL_BY_OPBEL'
      EXPORTING
        i_opbel   = gr_dfkkko->opbel
      TABLES
        tp_fkkopk = gt_fkkopk.

* Fetch the items in contract account document
    CALL FUNCTION 'FKK_REP_LINE_ITEM_SEL_BY_OPBEL'
      EXPORTING
        i_opbel   = gr_dfkkko->opbel
      TABLES
        pt_fkkopw = gt_fkkopw.

* Fetch the business partner items in contract account document
    CALL FUNCTION 'FKK_BP_LINE_ITEMS_SEL_BY_OPBEL'
      EXPORTING
        i_opbel  = gr_dfkkko->opbel
      IMPORTING
        e_count  = gv_count
      TABLES
        pt_fkkop = gt_fkkop.

    LOOP AT gt_fkkop REFERENCE INTO gr_fkkop.

* Set new abrzu and abrzo in the gt_fkkop if the record is exist
* in the table gt_items...
      READ TABLE gt_items REFERENCE INTO gr_items
        WITH KEY opbel = gr_fkkop->opbel
                 opupw = gr_fkkop->opupw
                 opupk = gr_fkkop->opupk
                 opupz = gr_fkkop->opupz
        BINARY SEARCH.

      IF sy-subrc EQ 0.

        APPEND INITIAL LINE TO gt_fkkop_new REFERENCE INTO gr_fkkop_new.

        MOVE-CORRESPONDING gr_fkkop->* TO gr_fkkop_new->*.

        gr_fkkop_new->abrzu = gr_items->abrzu_n.
        gr_fkkop_new->abrzo = gr_items->abrzo_n.
      ELSE.

        DELETE gt_fkkop INDEX sy-tabix.
      ENDIF.
    ENDLOOP.

    CHECK gv_count GT 0.

    gv_updated = abap_on.

* Update the document...
    CALL FUNCTION 'FKK_CHANGE_DOC_DB'
      EXPORTING
        i_fkkko       = gs_fkkko
        i_update_task = 'U'
      TABLES
        t_fkkop       = gt_fkkop_new
        t_fkkopk      = gt_fkkopk
        t_fkkopw      = gt_fkkopw
        t_fkkop_old   = gt_fkkop.

* Release the document...
    CALL FUNCTION 'FKK_DEQ_OPBELS_AFTER_CHANGES'.
  ENDLOOP.

  IF gv_updated IS NOT INITIAL.
    MESSAGE text-019 TYPE 'S'.
  ENDIF.
ENDFORM. " UPDATE_DATA

FORM set_message USING pv_mode TYPE char1
                       pv_opbel TYPE opbel_kk.

* Set the message...
  CASE pv_mode.
    WHEN 'L'.

* The document has been locked by another user!...
      LOOP AT gt_items REFERENCE INTO gr_items WHERE opbel EQ pv_opbel.
        gr_items->message = text-015.
      ENDLOOP.

    WHEN 'S'.

* System failure!...
      LOOP AT gt_items REFERENCE INTO gr_items WHERE opbel EQ pv_opbel.
        gr_items->message = text-018.
      ENDLOOP.
    WHEN OTHERS.

* The document header is not found!...
      LOOP AT gt_items REFERENCE INTO gr_items WHERE opbel EQ pv_opbel.
        gr_items->message = text-017.
      ENDLOOP.
  ENDCASE.
ENDFORM. " SET_MESSAGE

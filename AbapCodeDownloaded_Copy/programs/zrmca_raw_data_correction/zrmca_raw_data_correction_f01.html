FORM get_vkont_gpart USING pv_vtref TYPE vtref_kk
                           pv_cnum  TYPE zz_calling_num_kk
                  CHANGING pv_gpart TYPE gpart_ci_kk
                           pv_vkont TYPE vkont_ci_kk
                           pv_ppacc TYPE ppacc_kk.

  DATA: lv_telnum   TYPE ist_tnum,
        lv_areacode TYPE ist_acode.

  CHECK pv_vtref IS NOT INITIAL OR pv_cnum IS NOT INITIAL.

* Find correct Cont. Acc / BP based on reference contract...
  IF pv_vtref IS NOT INITIAL.

    IF selm EQ gc_raw AND p_prep IS NOT INITIAL.

      SELECT SINGLE vkont ppacc INTO (pv_vkont, pv_ppacc)
        FROM dfkk_vt_i
        WHERE vtkey EQ pv_vtref
        AND   valto EQ gc_valid_to.
    ENDIF.

    IF pv_vkont IS INITIAL.

      SELECT SINGLE vkont INTO pv_vkont
        FROM ist_tdata
        WHERE order_refnr EQ pv_vtref
        AND   valid_to    EQ gc_valid_to.
    ENDIF.

    IF pv_vkont IS NOT INITIAL.

      SELECT SINGLE gpart INTO pv_gpart FROM fkkvkp
        WHERE vkont EQ pv_vkont.
    ENDIF.
  ENDIF.

* IF Cont Acc / BP is not found, try to find it by MSISDN...
  CHECK ( pv_vkont IS INITIAL OR pv_gpart IS INITIAL ) AND pv_cnum IS NOT INITIAL.

  lv_telnum = pv_cnum.

  SELECT SINGLE vkont INTO pv_vkont
    FROM ist_tdata
    WHERE area_code   EQ lv_areacode
    AND   telnum      EQ lv_telnum
    AND   valid_to    EQ gc_valid_to.

  IF pv_vkont IS NOT INITIAL.

    SELECT SINGLE gpart INTO pv_gpart FROM fkkvkp
      WHERE vkont EQ pv_vkont.
  ENDIF.
ENDFORM. " GET_VKONT_GPART

    EXPORTING
      bitpackuuid    = gr_bit->bitpackuuid

    MOVE-CORRESPONDING gr_bit0_it->* TO lr_bit0_it_old->*.

        it_bit0_it = lt_bit0_it.

        it_bit0_it_all_upd = lt_bit0_it

      CHANGING

FORM find_cont_acc .
  DATA: lv_total_vkont TYPE zrmca_nofca,
        lv_job_counter TYPE char1,
        lv_temp        TYPE betrw_kk,
        lv_nofcontacc  TYPE zrmca_nofca,
        lv_index_low   TYPE syindex,
        lv_index_high  TYPE syindex,
        lv_vkont_low   TYPE vkont_kk,
        lv_vkont_high  TYPE vkont_kk.

  SELECT vkont INTO TABLE gt_vkont
    FROM dfkkinvdoc_h
    WHERE invdocno  IN s_bill
    AND gpart       IN s_gpart
    AND vkont       IN s_vkont
    AND crname      IN s_crname
    AND inv_process IN s_invprc
    AND mdcat       EQ p_mdcat
    AND doctype     IN s_docty
    AND budat       EQ p_budat
    AND crdate      IN s_crdate
    AND simulated   EQ co_true.

  IF sy-subrc NE 0.

    MESSAGE text-001 TYPE 'I'.
  ELSE.

    SORT gt_vkont BY vkont.

    DELETE ADJACENT DUPLICATES FROM gt_vkont.

* Find the number of contract account and divide them to four.
    lv_total_vkont = lines( gt_vkont ).
    lv_temp        = lv_total_vkont / gc_nofpp.
    lv_nofcontacc  = ceil( lv_temp ).

    lv_job_counter = lv_index_low  = 1.
    lv_index_high  = lv_nofcontacc.

* Create 4 parallel processes and submit itself by setting the field p_call = 'X'
    DO gc_nofpp TIMES.

      READ TABLE gt_vkont REFERENCE INTO gr_vkont INDEX lv_index_low.
      lv_vkont_low = gr_vkont->vkont.

      IF lv_index_high GT lv_total_vkont.
        lv_index_high = lv_total_vkont.
      ENDIF.

      READ TABLE gt_vkont REFERENCE INTO gr_vkont INDEX lv_index_high.
      lv_vkont_high = gr_vkont->vkont.

      PERFORM submit_prg USING lv_job_counter lv_vkont_low lv_vkont_high.

      IF lv_index_high EQ lv_total_vkont.
        EXIT.
      ENDIF.

      ADD: lv_nofcontacc TO lv_index_low,
           lv_nofcontacc TO lv_index_high,
           1             TO lv_job_counter.
    ENDDO.
  ENDIF.
ENDFORM. " FIND_CONT_ACC

FORM delete_sim_invdoc .

* Init protocoll
  CALL FUNCTION 'ARCHIVE_PROTOCOL_INIT'
    EXPORTING
      i_detailprotocol = p_prot.

* Message for test mode

    CALL FUNCTION 'ARCHIVE_PROTOCOL_COLLECT'
      EXPORTING
        i_msgtype = '1'
        i_msgid   = 'FKKINV'
        i_msgno   = '511'.
    IF 1 = 2. MESSAGE i511(fkkinv). ENDIF.
  ENDIF.

* Open cursor for DFKKINVDOC_H
  OPEN CURSOR WITH HOLD v_invdoc_cursor
    FOR SELECT * FROM dfkkinvdoc_h AS h
      WHERE invdocno  IN s_bill
      AND gpart       IN s_gpart
      AND vkont       IN s_vkont
      AND crname      IN s_crname
      AND inv_process IN s_invprc
      AND mdcat       EQ p_mdcat
      AND doctype     IN s_docty
      AND budat       EQ p_budat
      AND crdate      IN s_crdate
      AND simulated   EQ co_true.

  WHILE lv_eof EQ co_false.
* select records from database
    FETCH NEXT CURSOR v_invdoc_cursor INTO TABLE it_dfkkinvdoc_h
                                    PACKAGE SIZE co_max_read.

    IF sy-subrc NE 0.
*     there are no more entries in the database; end processing
      lv_eof = co_true.
    ELSE.

*     creation date must be older than the retention period for archiving
*     and no simulated invoicing documents should be archived
*      DELETE it_dfkkinvdoc_h WHERE crdate >= vd_invdoc.

*     analyse the header of the invoicing document
      PERFORM analyse_sim_invdocs CHANGING it_dfkkinvdoc_h.

      PERFORM delete_rest_sim_invdocs USING it_dfkkinvdoc_h
                                            p_delprd.

      PERFORM delete_sim_invdocs USING it_dfkkinvdoc_h
                                       lv_cnt_doc_tot
                                       p_delprd.


      IF it_dfkkinvdoc_h IS INITIAL.
        CONTINUE.
      ENDIF.

      IF p_delprd = co_true.
        CALL FUNCTION 'DB_COMMIT'.
      ENDIF.
    ENDIF.

  ENDWHILE.

  CLOSE CURSOR v_invdoc_cursor.

  CALL FUNCTION 'ARCHIVE_PROTOCOL_COLLECT'
    EXPORTING
      i_msgtype = '1'
      i_msgid   = 'FKKINV'
      i_msgno   = '532'
      i_msgv1   = lv_cnt_doc_tot.

  IF 1 = 2. MESSAGE i532(fkkinv). ENDIF.

* Call detailed protocol
  CALL FUNCTION 'ARCHIVE_PROTOCOL_WRITE'.
ENDFORM. " DELETE_SIM_INVDOC

FORM nrr_warning_email .

  SELECT * FROM zsd_nr_mail INTO TABLE gt_mail.

FORM sendmail.

  CONSTANTS:     gc_obj_ty  TYPE so_obj_tp    VALUE 'HTM',
                 gc_sender  TYPE ad_smtpadr   VALUE 'no-reply@albtelecom.al'.

  DATA:  lv_ok        TYPE so_text001,
         lv_mess_text TYPE text132,
         ex_bcs       TYPE REF TO cx_bcs,
         lv_error     TYPE char1.

  TRY.

      send_request = cl_bcs=>create_persistent( ).

      document = cl_document_bcs=>create_document(
                    i_type    = gc_obj_ty
                    i_text    = gt_text
                    i_subject = '' ).
      DATA : v_sub(100)  TYPE c VALUE 'Serial nr Range of SAP SD / Vargu i nr Serial ne SAP SD'.
      subject = v_sub.
      CALL METHOD send_request->set_message_subject
        EXPORTING
          ip_subject = subject.
      send_request->set_document( document ).
      sender = cl_cam_address_bcs=>create_internet_address( gc_sender ).
      send_request->set_sender( sender ).


      LOOP AT gt_mail REFERENCE INTO gr_mail WHERE mailadr IS NOT INITIAL.

        recipient = cl_cam_address_bcs=>create_internet_address( gr_mail->mailadr ).
        send_request->add_recipient( i_recipient = recipient ).
      ENDLOOP.

      send_request->set_send_immediately( 'X' ).
      lv_ok = send_request->send( ).

      IF lv_ok = abap_true.

        COMMIT WORK.
        SUBMIT rsconn01 WITH mode EQ 'INT' AND RETURN.
      ENDIF.

    CATCH cx_address_bcs cx_send_req_bcs cx_bcs INTO ex_bcs.

      lv_error = 'X'.
  ENDTRY.

  IF lv_error IS INITIAL.
    UPDATE zsd_nr_mail SET sent = 'X'.
  ENDIF.
ENDFORM. " SENDMAIL
